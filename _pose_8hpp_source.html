224  Pose deltaRelative(double t) const {
225  // twist = (x', y', h')
226  // dh(world) = h' * dt
227  // dx(world) = dx(local)cos(dh(world)) - dy(local)sin(dh(world))
228  // = integral(x'cos(h't) - y'sin(h't), dt)
229  // = x'/h' sin(h't) + y'/h' (cos(h't) - 1)
230  // dy(world) = dx(local)sin(dh(world)) + dy(local)cos(dh(world))
231  // = integral(x'sin(h't) + y'cos(h't), dt)
232  // = x'/h' (1 - cos(h't)) + y'/h' sin(h't)
233  double vx = linear().x();
234  double vy = linear().y();
235  double vh = angular();
236 
237  // From above: sin(h't)/h' and (1 - cos(h't))/h' respectively
238  double sine_frac, cosine_frac;
239 
240  if (std::abs(vh) < 1e-6) {
241  // Small-angle approximations
242  // sin(h't) ~ h't
243  // 1 - cos(h't) ~ 0
244  sine_frac = t;
245  cosine_frac = 0;
246  } else {
247  sine_frac = std::sin(vh * t) / vh;
248  cosine_frac = (1 - std::cos(vh * t)) / vh;
249  }
250 
251  return Pose(vx * sine_frac - vy * cosine_frac,
252  vx * cosine_frac + vy * sine_frac, vh * t);
253  }
